public class Room {
    private static Room instance;
    private String roomNum;
    private String roomName;
    private ArrayList<String> roomDescription;
    private ArrayList<Exit> exits;
    private ArrayList<Item> items;
    private boolean visited;

    private Room(String roomNum, String roomName, ArrayList<String> roomDescription, ArrayList<Exit> exits, ArrayList<Item> items) {
        this.roomNum = roomNum;
        this.roomName = roomName;
        this.roomDescription = roomDescription;
        this.exits = exits;
        this.items = items;
        this.visited = false;
    }

    public static Room getInstance(String roomNum, String roomName, ArrayList<String> roomDescription, ArrayList<Exit> exits, ArrayList<Item> items) {
        if (instance == null) {
            instance = new Room(roomNum, roomName, roomDescription, exits, items);
        }
        return instance;
    }
